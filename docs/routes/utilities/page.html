Util
Methods for various use cases
        
  ${`import { wfcUtil } from '@webformula/material/services';
  
  // --- Below are some of the more useful methods ---

  // throttle function calls by milliseconds
  wfcUtil.throttle(() => {}, 200);

  wfcUtil.debounce(() => {}, 200);

Date
Wrapper class around Date and Intl.DateTimeFormat, used by date pickers
        
  ${`import { wfcDate } from '@webformula/material/services';
  
  // --- Below are some of the more useful methods ---

  // Set local for anything using this util, including date pickers
  wfcDate.locale = 'EN-US';

  // Set timezone for anything using this util, including date pickers
  wfcDate.timeZone = 'GMT';

  // Parse accepts strings (yyyy-mm-dd) and Date objects
  // returns Date object
  wfcDate.parse('2023-01-20');

  // build on top of Intl.DateTimeFormat, using the utils locale and timezone
  // returns String
  // ddd, MMM DD = Thu, Dec 12
  // YYYY-MMMM-dddd = 2019-month-wednesday
  // YY/MMM/ddd = 19/10/22
  wfcDate.format(new Date(), 'yyyy-mm-dd');

  // returns Date object
  wfcDate.buildFromParts({ year: '2023', month: '01', day: '20' });

  wfcDate.getYear(new Date());
  wfcDate.getMonth(new Date());
  wfcDate.getWeekDay(new Date());
  wfcDate.getMonthDay(new Date());
  wfcDate.getFirstDateOfMonth(new Date());
  wfcDate.getParts(new Date()); // { year, month, day }

  wfcDate.getMonthNames();
  wfcDate.getMonthNamesShort();
  wfcDate.getDayNames('long | short | narrow'); // mon, tue, wed, thur, fri

  // Add to individual parts of date, use negative numbers to subtract
  // returns Date object
  wfcDate.addToDateByParts(new Date(), { year: 1, month: 1, day: 1 });

  // Set individual parts of date
  // returns Date object
  wfcDate.setDateByParts(new Date(), { year: 1, month: 1, day: 1 });`}
