  // --- Below are some of the more useful methods ---

  // throttle function calls by milliseconds
  mdwUtil.throttle(() => {}, 200);

Date
Wrapper class around Date and Intl.DateTimeFormat, used by date pickers
        
  ${`import { mdwDate } from '@webformula/material/services';
  
  // --- Below are some of the more useful methods ---

  // Set local for anything using this util, including date pickers
  mdwDate.locale = 'EN-US';

  // Set timezone for anything using this util, including date pickers
  mdwDate.timeZone = 'GMT';

  // Parse accepts strings (yyyy-mm-dd) and Date objects
  // returns Date object
  mdwDate.parse('2023-01-20');

  // build on top of Intl.DateTimeFormat, using the utils locale and timezone
  // returns String
  // ddd, MMM DD = Thu, Dec 12
  // YYYY-MMMM-dddd = 2019-month-wednesday
  // YY/MMM/ddd = 19/10/22
  mdwDate.format(new Date(), 'yyyy-mm-dd');

  // returns Date object
  mdwDate.buildFromParts({ year: '2023', month: '01', day: '20' });

  mdwDate.getYear(new Date());
  mdwDate.getMonth(new Date());
  mdwDate.getWeekDay(new Date());
  mdwDate.getMonthDay(new Date());
  mdwDate.getFirstDateOfMonth(new Date());
  mdwDate.getParts(new Date()); // { year, month, day }

  mdwDate.getMonthNames();
  mdwDate.getMonthNamesShort();
  mdwDate.getDayNames('long | short | narrow'); // mon, tue, wed, thur, fri

  // Add to individual parts of date, use negative numbers to subtract
  // returns Date object
  mdwDate.addToDateByParts(new Date(), { year: 1, month: 1, day: 1 });

  // Set individual parts of date
  // returns Date object
  mdwDate.setDateByParts(new Date(), { year: 1, month: 1, day: 1 });`}
